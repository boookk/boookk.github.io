class Node:
    def __init__(self, data):
        self.data = data
        self.next = None


class Stack:
    def __init__(self):
        self.head = None

    def push(self, item):
        new_head = Node(item)
        new_head.next = self.head
        self.head = new_head

    def pop(self):
        if self.isEmpty():
            return "Stack is Empty"
        item = self.head
        self.head = self.head.next
        return item

    def peek(self):
        if self.isEmpty():
            return "Stack is Empty"
        return self.head.data

    def isEmpty(self):
        return self.head is None
